Enum match_status_enum {
  UNSCHEDULED 
  SCHEDULED
  IN_PROGRESS
  FINISHED
  CANCELLED
}

Table Championship {
  championship_id integer [primary key]
  championship_name varchar(80) [not null]
  championship_description varchar(500) [not null]
  championship_startDate timestamp [not null]
  championship_endDate timestamp [null]
  championship_minTeams integer [not null]
  championship_maxTeams integer [not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
}

Table Team {
  team_id integer [primary key]
  team_name varchar(50) [not null]
  team_acronym varchar(3) [not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
}

Table Match {
  match_id integer [primary key]
  match_championshipId integer [not null]
  match_homeTeamId integer [not null]
  match_visitingTeamId integer [not null]
  match_datetime timestamp [not null]
  match_location varchar(150) [not null]
  match_status match_status_enum [not null]
  match_homeTeamPontuation integer [null]
  match_visitingTeamPontuation integer [null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
  indexes {
    (match_championshipId, match_datetime)
  }
}
